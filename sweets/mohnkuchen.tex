\begin{recipe}{Mohnkuchen}
    \index{Mohnkuchen}
    \index{Kuchen!Mohn-}
    \ingredient{Mohn}
    \region{Österreich}
    \source{Mama}

    \begin{ingredients}
        \entry[120\gram]{geriebener Mohn}
        \entry[120\gram]{geriebene Nüsse}
        \entry[120\gram]{Zucker}
        \entry[120\gram]{Öl}
        \entry[4]{Eier}
        \entry{Zitronenschale}
        \entry{etwas Rum oder Milch}
    \end{ingredients}

    \begin{instructions}
        Schnee aus den 4 Eiern herstellen.
        Restliche Zutaten mischen und den Schnee unterheben.
        Ca.\ 35--40 Minuten bei mittlerer Hitze backen.

        Marmellade enfüllen und eventuell mit Zitronenglasur glasieren.
    \end{instructions}
\end{recipe}
