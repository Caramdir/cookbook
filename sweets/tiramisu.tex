\begin{recipe}[\vegetarian]{Tiramisu}
    \index{Tiramisu}%
    \region{Italien}%
    \source{Claudia}%

    \begin{ingredients}
        \entry[500\gram]{Mascarpone}%
        \entry[4]{Eidotter}%
        \entry[100\gram]{Zucker}%
        \entry{Biskotten}%
        \entry{Cointreau}
        \entry{Kakaopulver}
        \entry{Kaffee (stark!)}
        \entry{geriebene Orangenschale}
    \end{ingredients}

    \begin{instructions}
        Zucker und Dotter 20\minutes schaumig rühren.
        Orange und Mascarpone hinzufügen.
        Aufschichten.

        8 Stunden stehen lassen.
    \end{instructions}
\end{recipe}
