\begin{recipe}{Chocolate Brownies}
    \index{Chocolate Brownies}
    \index{Brownies}
    \ingredient{Schokolade}
    \source{\href{https://www.bbc.co.uk/food/recipes/richchocolatebrownie_1933}{BBC Food}}

    \begin{ingredients}
        \entry[135\gram]{butter}
        \entry[270\gram]{caster sugar}
        \entry[84\gram]{dark chocolate, broken into pieces}
        \entry[5]{medium eggs}
        \entry[66\gram]{plain flour}
        \entry[33\gram]{cocoa powder}
    \end{ingredients}

    \begin{instructions}
        Heat the oven to 190\degC/170\degC Fan/Gas 5. Line a 20x18\cm baking tin with baking paper.

        Gently melt the butter and the sugar together in a large pan. Once melted, take off the heat and add the chocolate. Stir until melted.

        Beat in the eggs, then stir in the rest of the ingredients.

        Pour into the prepared tin and bake for 30â€“40 minutes, or until the top of the brownie is firm but the inside still feels soft.

        Take out of the oven and cool in the tin. Cut into 5\cm squares when cool.
    \end{instructions}
\end{recipe}
