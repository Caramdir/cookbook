\begin{recipe}[\vegetarian]{Zimtblätter}
    \index{Zimtblätter}
    \index{Kekse!Zimtblätter}
    \source{Mama}
    \region{Österreich}

    \begin{ingredients}
        \entry[\half\kg]{glattes Mehl}
        \entry[\quarter\kg]{Butter}
        \entry[\quarter\kg]{Staubzucker}
        \entry[2]{Eier}
        \entry[1]{Dotter}
        \entry{Schale von einer halben Zitrone}
        \entry[\quarter\kg]{fein geriebene Haselnüsse}
        \entry[1\tl]{Vanillezucker}
        \entry[1\tl]{Zimt}
        \entry[]{Prise Salz}
    \end{ingredients}

    \begin{instructions}
        Teig kneten.
        In Folie wickeln und kalt stellen.

        Backrohr auf ca.\ 200\degC.
        Kekse auf ein bemehltes Backblech und ca.\ 10--12\minutes\ backen.
    \end{instructions}
\end{recipe}
