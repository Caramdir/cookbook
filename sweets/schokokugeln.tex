\begin{recipe}[\vegetarian]{Schokokugeln (Rumkugeln)}
    \index{Schokokugeln}
    \index{Rumkugeln}
    \index{Kekse!Schokokugeln}
    \index{Kekse!Rumkugeln}
    \ingredient{Schokolade}
    \source{Mama}

    \begin{ingredients}
        \entry[200\gram]{Schokolade}
        \entry[100\gram]{Zucker}
        \entry[100\gram]{Magarine}
        \entry[2\tl]{Kakao}
        \entry[2\tl]{Rum}
        \entry[1]{Dotter}
        \entry[2\pkg]{Vanillezucker}
    \end{ingredients}

    \begin{instructions}
        Schokolade und Magarine im Wasserbad erweichen, glattrühren und mit anderen Zutaten mischen.
        Kaltstellen.
        Kugeln formen und in Streusel wälzen.
    \end{instructions}
