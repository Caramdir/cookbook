\begin{recipe}{Wiener Kartoffelsuppe}
    \index{Wiener Kartoffelsuppe}
    \index{Kartoffelsuppe}
    \ingredient{Erdäpfel}
    \region{Österreich}
    \source{Mama}

    \begin{ingredients}
        \entry[300\gram]{rohe Kartoffel in kleinen Würfeln}
        \entry[1\half\liter]{Suppe}
        \entry[40\gram]{Fett}
        \entry[50\gram]{Selchspeck in kleinen Würfeln}
        \entry[1]{Zwiebel, fein geschnitten}
        \entry{Wurzelwerk, geschnitten}
        \entry{einige getrocknete, vorgeweichte Pilze}
        \entry{etwas Zitronenschale}
        \entry[20\gram]{Mehl}
        \entry{Salz, Pfeffer, Kümmel, Majoran}
        \entry[3--4\el]{Rahm}
        \entry{Petersilie}
    \end{ingredients}

    \begin{instructions}
        Speck in Fett rösten, dazu Zwiebel und Wurzelwerk.
        Mit Mehl stauben, mit Suppe aufgießen.
        Gewürze, Pilze, Kartoffel dazu.
        Alles weichkochen.
        Nachwürzen und Rahm einrühren.
    \end{instructions}
\end{recipe}
