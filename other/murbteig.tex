\begin{recipe}{Mürbteig}
    \label{murbteig}%
    \index{Murbteig@Mürbteig}%
    \index{Teig}%

    \begin{ingredients}
        \entry[300\gram]{Mehl}
        \entry[200\gram]{Butter}
        \entry[100\gram]{Staubzucker}
        \entry[2]{Eier}
        \entry{Milch}
    \end{ingredients}

    \begin{instructions}
        Mehl, Butter, Zucker und Eier zusammenkneten.
        Falls der Teig zu trocken ist, etwas Milch hinzufügen.
        Den Teig mindestens eine halbe Stunde kalt rasten lassen.
    \end{instructions}
\end{recipe}
