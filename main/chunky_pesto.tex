\begin{recipe}[\vegan]{Chunky Pesto}
    \label{chunky_pesto}
    \index{Chunky Pesto}
    \index{pesto}
    \source{New York Times \cite{nyt-fazzoletti-chunky-pesto, nyt-minimalist-pasta-in-the-rought}}
    \cf{fazzoletti}%
    \yield{4 servings}

    \begin{ingredients}
        \entry[2\cups]{loosely packed basil leaves, rinsed and dried}
        \entry[\half]{clove garlic, peeled, or more to taste}
        \entry[\half\cup]{extra virgin olive oil, or more as desired}
        \entry[\quarter\cup]{cup pine nuts}
        \entry[\half\cup]{grated Parmesan}
        \entry{salt} 
    \end{ingredients}

    \begin{instructions}
        In a food processor or blender, combine the basil with a pinch of salt, the garlic and about half the oil.
        Process, stopping to scrape down the sides of the container if necessary, and adding the rest of the oil gradually.
        Add the nuts and cheese, and pulse a few times.
        The pesto should be well combined but still chunky. 
    \end{instructions}
\end{recipe}
