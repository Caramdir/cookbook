\begin{recipe}{Eiernockerln}%
    \index{Eiernockerln}%
    \index{Nockerln!Eier-}%
    \region{Österreich}%
    \source*{oesterreichische-kueche-nach-rokitansky}%
    \yield{4 Portionen}%

    \begin{ingredients}
        \entrysection{Nockerlteig}
        \entry[300\gram]{Mehl}
        \entry[30\gram]{Butter}
        \entry{Salz}
        \entry{Muskat}
        \entry[1]{Ei}
        \entry{Milch}
        \entrysection{}
        \entry[50\gram]{Butter}
        \entry[2]{Eier}
        \entry[2\el]{Sauerrahm}
        \entry{Petersilie}
    \end{ingredients}

    \begin{instructions}
        In einer Pfanne lässt man Butter aufschäumen, schwenkt darin die gekochten Nockerln, würzt min Salz und einer Prise Muskat und lässt die Nockerln ankrusten.
        Die 3 Eier mit 2 Esslöffel Sauerrahm versprudeln, über die Nockerln leeren, kurz durchmischen und in der Pfanne goldbraun anbacken.
        Beim Anrichten mit gehackter Petersilie bestreuen.
    \end{instructions}
\end{recipe}
